create or replace NONEDITIONABLE PACKAGE BODY P_SECTION6 AS

 /* =========== SECTION 6-1 일반변수 / ROWTYPE ============ */ 
    -- SECTION 6-1
    FUNCTION PF_GET_NAME1
        (
           P_CST_ID IN CST_INFO.CST_ID%TYPE
        ) RETURN VARCHAR2
        AS
            V_NAME CST_INFO.NAME%TYPE;
        BEGIN
            SELECT NAME
            INTO V_NAME
            FROM CST_INFO
            WHERE CST_ID = P_CST_ID;
        
            RETURN V_NAME;
        END PF_GET_NAME1;

    -- SECTION 6-1     
    FUNCTION PF_GET_NAME2
     (
       P_CST_INFO IN CST_INFO%ROWTYPE
     ) RETURN CST_INFO%ROWTYPE
    AS
      R_CST_INFO CST_INFO%ROWTYPE;
    BEGIN
        SELECT *
        INTO R_CST_INFO
        FROM CST_INFO
        WHERE CST_ID = P_CST_INFO.CST_ID;
        
       RETURN R_CST_INFO;
    END PF_GET_NAME2;  
    
    -- SECTION 6-1
    PROCEDURE PSP_GET_NAME1
    (
       P_CST_ID IN CST_INFO.CST_ID%TYPE,
       V_NAME  OUT CST_INFO.NAME%TYPE    
    ) 
    AS   
    BEGIN
        SELECT NAME
        INTO V_NAME
        FROM CST_INFO
        WHERE CST_ID = P_CST_ID;
    END PSP_GET_NAME1;
    
   -- SECTION 6-1 
   PROCEDURE PSP_GET_NAME2
    (
       P_CST_INFO IN CST_INFO%ROWTYPE,
       R_CST_INFO OUT CST_INFO%ROWTYPE
    ) 
    AS
    BEGIN
        SELECT *
        INTO R_CST_INFO
        FROM CST_INFO
        WHERE CST_ID = P_CST_INFO.CST_ID;
    END PSP_GET_NAME2;

    -- SECTION 6-1
    PROCEDURE PSP_SECTION6_1 
    AS 
      -- 일반변수
      P_CST_ID CST_INFO.CST_ID%TYPE;
      V_NAME CST_INFO.NAME%TYPE;
      -- ROWTYPE 변수
      P_CST_INFO CST_INFO%ROWTYPE;
      R_CST_INFO CST_INFO%ROWTYPE;
    BEGIN
      -- FUNCTION 호출 
      P_CST_ID := 'C005';
      -- 일반변수 이용
      V_NAME := P_SECTION6.PF_GET_NAME1(P_CST_ID);
      DBMS_OUTPUT.PUT_LINE('V_NAME > ' || V_NAME);
      -- ROWTYPE 이용
      SELECT *
      INTO P_CST_INFO
      FROM CST_INFO
      WHERE CST_ID = P_CST_ID;
      
      R_CST_INFO := P_SECTION6.PF_GET_NAME2(P_CST_INFO);
      DBMS_OUTPUT.PUT_LINE('R_CST_INFO.NAME > ' || R_CST_INFO.NAME);
      DBMS_OUTPUT.PUT_LINE(CHR(10) || CHR(13));
    
      -- PROCEDURE 호출 
      P_CST_ID := 'C001';
      -- 일반변수 이용
      P_SECTION6.PSP_GET_NAME1(P_CST_ID, V_NAME);
      DBMS_OUTPUT.PUT_LINE('SP V_NAME > ' || V_NAME);
      -- ROWTYPE 이용
      SELECT *
      INTO P_CST_INFO
      FROM CST_INFO
      WHERE CST_ID = P_CST_ID;
      P_SECTION6.PSP_GET_NAME2(P_CST_INFO, R_CST_INFO);
      DBMS_OUTPUT.PUT_LINE('SP R_CST_INFO.NAME > ' || R_CST_INFO.NAME);
  END PSP_SECTION6_1;
  
  /* =========== SECTION 6-3 RECORD ============ */ 
  FUNCTION PF_GET_RECORD
  (
    P_CST_ID CST_INFO.CST_ID%TYPE
  )RETURN CST_INFO_REC_TYP
  AS
    REC_CST_INFO CST_INFO_REC_TYP;
  BEGIN
    SELECT NAME, BIRTH, MOBILE, '커피' AS FAVORITE
    INTO REC_CST_INFO
    FROM CST_INFO
    WHERE CST_ID = P_CST_ID
    ;
    RETURN REC_CST_INFO;
  END PF_GET_RECORD
  ;
  PROCEDURE PSP_GET_RECORD
  (
    P_CST_ID CST_INFO.CST_ID%TYPE,
    REC_CST_INFO OUT CST_INFO_REC_TYP
  )
  AS
  BEGIN
    SELECT NAME, BIRTH, MOBILE, '커피' AS FAVORITE
    INTO 
        REC_CST_INFO.V_NAME, REC_CST_INFO.V_BIRTH,
        REC_CST_INFO.V_MOBILE, REC_CST_INFO.V_FAVORITE
--    REC_CST_INFO --(위의 4개를 생략하고 이렇게 써도 값이 들어간다.)
    FROM CST_INFO
    WHERE CST_ID = P_CST_ID
    ;
  END PSP_GET_RECORD
  ;
  
  PROCEDURE PSP_SECTION6_3
  AS
  -- 일반변수
     P_CST_ID CST_INFO.CST_ID%TYPE;
  -- 레코드 타입 변수
     CST_INFO_REC CST_INFO_REC_TYP;
  BEGIN
    P_CST_ID := 'C001';
    CST_INFO_REC := P_SECTION6.PF_GET_RECORD(P_CST_ID);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_NAME: ' || CST_INFO_REC.V_NAME);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_BIRTH: ' || CST_INFO_REC.V_BIRTH);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_MOBILE: ' || CST_INFO_REC.V_MOBILE);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_FAVORITE: ' || CST_INFO_REC.V_FAVORITE);
    
    DBMS_OUTPUT.PUT_LINE(CHR(10) || CHR(13));
    
    P_CST_ID := 'C003';
    P_SECTION6.PSP_GET_RECORD(P_CST_ID, CST_INFO_REC);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_NAME: ' || CST_INFO_REC.V_NAME);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_BIRTH: ' || CST_INFO_REC.V_BIRTH);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_MOBILE: ' || CST_INFO_REC.V_MOBILE);
    DBMS_OUTPUT.PUT_LINE('CST_INFO_REC.V_FAVORITE: ' || CST_INFO_REC.V_FAVORITE);
    
    
  END PSP_SECTION6_3
  ;
  
  
END P_SECTION6;
