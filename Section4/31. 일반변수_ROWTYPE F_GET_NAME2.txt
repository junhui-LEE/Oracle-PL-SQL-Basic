create or replace NONEDITIONABLE FUNCTION F_GET_NAME2
(
    P_CST_INFO CST_INFO%ROWTYPE
)RETURN CST_INFO%ROWTYPE
AS
    R_CST_INFO CST_INFO%ROWTYPE;
BEGIN
    SELECT * 
    INTO R_CST_INFO
    FROM CST_INFO
    WHERE CST_ID = P_CST_INFO.CST_ID
    ;
    RETURN R_CST_INFO;
END F_GET_NAME2;
