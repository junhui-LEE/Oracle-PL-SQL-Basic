create or replace NONEDITIONABLE PACKAGE P_COMM AS
    /* 패키지 내부 전역 레코드 */
    TYPE CST_INFO_REC_TYP IS RECORD (
      V_NAME     CST_INFO.NAME%TYPE,
      V_BIRTH    CST_INFO.BIRTH%TYPE,
      V_MOBILE   CST_INFO.MOBILE%TYPE,
      V_FAVORITE VARCHAR2(100)
    );
     /* WEEK TYPE */
    TYPE RTN_CUR IS REF CURSOR;
    /* RECORD TYPE */
    TYPE REC_CUR IS REF CURSOR RETURN CST_INFO_REC_TYP;
    /* STRONG TYPE */
    TYPE CST_CUR_TYP IS REF CURSOR RETURN CST_INFO%ROWTYPE;
END P_COMM;
